00850                             }
00851                      }
00852               }
00853 
00854               // correct current column
00855               result.SetAt(istd::CIndex2d(hhIndex, hhIndex), -hhLength);
00856               for (int i = hhIndex + 1; i < Height; ++i){
00857                      result.SetAt(istd::CIndex2d(hhIndex, i), 0);
00858               }
00859        }
00860 
00861        return true;
